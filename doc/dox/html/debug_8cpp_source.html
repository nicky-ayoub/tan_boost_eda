108  string kw;
109  start = 0;
110  do {// This is the beginning of a do-while loop
111  // Look for a comma delimiting an internal keyword
112  // We are assuming well-formed multiple keywords -

296 /***********************************
297 * Debug::specify()
298 * Extracts options and keywords from a Debug specification string
299 * and deals with them "appropriately."
300 *
301 * PSEUDOCODE:
302 * look for an option
303 * while(there is an option)
304 * look for next option or end of string
305 * look for a keyword in the substring thus found
306 * while(where is a keyword)
307 * depending upon the option found
308 * save the keyword OR
309 * extract and save some value OR
310 * ignore the keyword
311 * look for a keyword
312 * end while
313 * end while
314 *
315 * Option strings have the form
316 * L:keyword,keyword,L:keyword,keyword,
317 * L:L:keyword,keyword
318 * L:L:,keyword
319 * etc. Terminating commas are optional
320 * Leading commas are harmless.
321 **************************************/
322 void Debug::specify(const char * opt) {
323  string s = opt;
324 
325  // Find options
326  unsigned int j = s.find_first_of(separator,0);
327  while(j < s.length()) { // Option found
328  // j points to the separator at the end of this option
329 
330  unsigned int k = s.find_first_of(separator,j+1);
331  // k now points to the separator at the end of the next option
332  // or to outer space
333 
334  string Option;
335  if(j == 0) {
336  Option = "0";
337  Option += s[0];
338  } else {
339  Option = string(s,j-1,2);
340  }
341 
342  // Find keywords
343  unsigned int l = s.find_first_of(',',j+1);
344 
345  // Deal with the case of no comma, like F:keywordD: or
346  // F:keyword
347  if((l >= k) && (j < k-2)) {
348  if( k > s.length() ) // No trailing option
349  l = s.length();
350  else l = k-1; // Trailing option is found
351  }
352  while(l < k) { // keyword found
353  string kw;
354  if(j+1 < l ) {
355  kw = string(s,j+1,l-j-1);
356  } else {
357  kw = "";
358  }
359 
360  if( l < s.length() ) {
361 
362  j = l; // Advance to virtual ( or real ) comma.
363  unsigned int m = s.find_first_of(',',l+1);
364 
365  // Again, what if there's no trailing comma in a
366  // sequence of keywords like F:abcd,efgh,ijklD:
367  // or F:a,b,c,d
368  if((m >= k) && (j < k-1)) {
369  if( (m > s.length()) && ( k > s.length()))
370  m = s.length();
371  else m = k-1;
372  }
373  l = m;
374  } else l = k; // Bail out of the loop
375 
376  // This is where we deal with the keywords and options
377  switch(Option[0]) {
378  case 'D': // Turn debugging on
379  if(kw.length()) keywords.push_back(kw);
380  break;
381 
382  case 'O': // Changing output, maybe
383 #ifdef STL96
384  if(!( (! kw.length()) || kw==filename)) {
385 #else
386  if(kw.length() && kw != filename) {
387 #endif
388  filename = kw;
389  if( kw == "-"){
390  pDebugStream=&cout;
391  }else{
392  if(kw == "--") {
393  pDebugStream=&clog;
394  }else{
395  if(NULL!=pFile){
396  delete pFile;
397  pFile=NULL;
398  }
399  pFile=new(ofstream);
400  pFile->open(filename.c_str(),ios::app);
401  pDebugStream=pFile;
402  }
403  }
404  }
405  break;
406 
407  case 'T': // Turn tracing on
408  if(kw.length()) functions.push_back(kw);
409  break;
410 
411  case 'M': // Turn interval timing on
412  if(kw.length()) timekeys.push_back(kw);
413  break;
414 
415  default: // Ignore other options silently
416  break;
417  }
418  } // end of loop on finding keywords
419  // This is where we complete dealing with options
420  switch(Option[0]) {
421  case 'D': // Turn debugging on
422  Debugging = true;
423  break;
424 
425  case 'd': // Turn debugging off
426  CLEAR(keywords);
427  Debugging = false;
428  break;
429 
430  case 'O': // Changing output, maybe - khtan : no longer needed
431  break;
432 
433  case 'T': // Turn tracing on
434  Tracing = true;
435  break;
436 
437  case 't': // Turn tracing off
438  CLEAR(functions);
439  Tracing = false;
440  break;
441 
442  case 'M': // Turn interval timing on
443  Timing = true;
444  break;
445 
446  case 'm': // Turn interval timing off
447  CLEAR(timekeys);
448  CLEAR(timers);
449  Timing = false;
450  break;
451 
452  default: // Ignore other options silently
453  break;
454  }
455  j = k; // Advance to next option
456  } // end of loop on finding options
457 }
