289 template<typename ContainerType>
290 bool DPassable(ContainerType& vDLogic, DLogic result){
291 /* Verifies that if D/_D is in the input set, the result
292  contains a D/_D.
293 
294  This is used to indicate whether the D/_D is being
295  sensitised from input to output, or stopped.
296 
297  If there is an X in the input, it is too early to tell, return is trivally true.
298  If there is no D/_D in the input, the return is trivially true.
299  If there is a D/_D in the input, the result is true only if the result is D/_D.
300  */
301  ContainerType::iterator target=std::find_if(vDLogic.begin(),vDLogic.end(),isXTypeFunctor());
302  bool bReturn=false;
303  if(vDLogic.end()==target){ // container has no Xs
304  target=std::find_if(vDLogic.begin(),vDLogic.end(),isDTypeFunctor());
305  if(vDLogic.end()!=target){ // container contains a D/_D
306  bReturn=(DLogic::D==result||DLogic::_D==result);
307  }else{
308  bReturn=true;
309  }
310  }else{
311  bReturn=true;
312  }
313  return bReturn;
314 };
315 bool OutputConsistent(DLogic sA, DLogic sB){
316 /* Verifies that the sA can replace sB without violating consistency rules.
317  where sA == signal evaluated for update
318  sB == signal already on output edges
319 
320  For example, if sA=1 and sB=0, there is inconsistency.
321  if sB=X, any value of sA is fine
322  if sB=D or _D, we are seeing a stuck-at-fault, so skip as well
323  */
324  if(DLogic::ONE==sB || DLogic::ZERO==sB ){
325  return (sA==sB);
326  }else{
327  return true;
328  }

331 DLogic EvaluateOutputs(Vertex& v, GraphType& g){
332  /* The functionality can be described as follows :
333  RealSignals, R=={ONE,ZERO,D,_D}
334  Since the vertex has 0 inputs, we are talking about ensuring that all
335  out_edges have the same signal value. This can then be used to
336  set the remaining out_edges
337  Set of out_edges | Evaluated result
338  -----------------|-----------------
339  R,X,X,X | R
340  R,R,X,X | R
341  R1,R2,X,X | assert error ( R1 != R2 )
342  X,X,X,X | X
343 
344  By convention,
345  D==good/bad==0/1 and _D==1/0
346  Should _D and 1 be considered compatible?
347  */
348  typedef boost::property_map<GraphType,boost::vertex_attribute_t>::type VertexAttrMapType;
349  typedef boost::property_map<GraphType,boost::edge_attribute_t>::type EdgeAttrMapType;
350  typedef boost::graph_traits<GraphType>::out_edge_iterator OutEdgeIteratorType;
351  Debug D("EvaluateOutputs");
352  VertexAttrMapType vMap=boost::get(boost::vertex_attribute,g);
353  EdgeAttrMapType eMap=boost::get(boost::edge_attribute,g);
354  string VertexLabel=vMap[v]["label"];
355  D.Dbg("1","vertex label==",VertexLabel);
356  OutEdgeIteratorType startEI,endEI;
357  set<string> RealSignalSet;
358  for( tie(startEI,endEI)=out_edges(v,g);startEI!=endEI;++startEI){
359  string& signal=eMap[*startEI]["label"];
360  D.Dbg("1","signal==",signal);
361  if("X"!=signal){
362  RealSignalSet.insert(signal);
363  }//if
364  }//for
365  string sResult="X";
366  switch(RealSignalSet.size()){
367  case 0: // bec no driving signal found
368  break;
369  case 1: // one consistent driving signal found
370  sResult=*(RealSignalSet.begin());
371  break;
372  default : // more than one incompatible driving signal
373  cout << "Error! More than one incompatible signal found on out_edges of " << VertexLabel << "\n";
374  break;
375  }//switch
376  D.Dbg("1","sResult==",sResult);
377  return DLogic(sResult);
378 };

515  template <class Vertex, class GraphType> DLogic suggestEnablingSignal(Vertex v, const GraphType & g){
516  /* Looks at source vertices of all incoming edges and collect
517  their func values. Return enabling signal, based on func values if
518  NAND,NOR,AND and OR found. Otherwise, no simple algorithm exists, so we
519  return DLogic::ONE
520  Refactor this function if more models have to be supported.
521  */
522  Debug D("getEnablingSignal");
523  set<string> setFunc;
524  // process source vertices, putting func in set
525  InEdgeIteratorType startEI,endEI;
526  Vertex vSource;
527  for(tie(startEI,endEI)=in_edges(v,g);startEI!=endEI;++startEI){
528  vSource=source(*startEI,g);
529  NodeHelper hSource(boost::get(vertex_attribute,g,vSource)["label"]);
530  setFunc.insert(hSource.getFunc());
531  }//for
532 
533  DLogic dResult=DLogic::ONE; // default
534  string& sFunc=const_cast<string&>(*(setFunc.begin()));
535  switch(setFunc.size()) {
536  case 0:
537  cout << "Warning: vertex drives no function model.\n";
538  break;
539  case 1:
540  if("nand"==sFunc || "and"==sFunc){
541  dResult=DLogic::ONE;
542  }else if("nor"==sFunc || "or"==sFunc){
543  dResult=DLogic::ZERO;
544  }
545  break;
546  default:
547  cout << "Warning: vertex drives multiple function models. Using default enabling signal.\n";
548  break;
549  }//switch
550  D.Dbg("1","dResult==",dResult.GetString());
551  return dResult;

684 void RunGraph<G>::seedDFrontier(DFrontierType& dF){
685 /* Search graph for edges with D/_D and put their target vertices
686  into DFrontier. Checks before inserting vertices so that DFrontier
687  has no duplicate vertices.
688  Note: Tradeoff performance to search for all faults in graph, so that
689  we can study multiple faults. Alternate implementation could return when
690  first fault found.
691 */
692  Debug D("seedDFrontier");
693  EdgeIteratorType firstEI,lastEI;
694  VertexType vTarget;
695  for(tie(firstEI,lastEI)=edges(_g);firstEI!=lastEI;++firstEI){
696  if(("D"==_e[*firstEI]["label"])||("_D"==_e[*firstEI]["label"])){
697  vTarget=target(*firstEI,_g);
698  DFrontierType::iterator iVertexFound=std::find(dF.begin(),dF.end(),vTarget);
699  if(dF.end()==iVertexFound){
700  dF.push_back(vTarget);
701  }//if vTarget not in DFrontier
702  }//if edge has fault injected
703  }//for all edges
704  dumpDFrontier(dF,_v);
705 };

802 tripleBool RunGraph<G>::propagateChange(VertexType v, G& g,PropagateContainerType& cv) {
803  Debug D("propagateChange");
804  string VertexLabel=_v[v]["label"];
805  D.Dbg("1","vertex label==",VertexLabel);
806  NodeHelper VertexHelper(VertexLabel);
807  const string& VertexFunc=VertexHelper.getFunc();
808  DLogic outSignal,currentSignal;
809 
810  bool bOutputFound=false;
811  bool bInconsistentOutput=false;
812  bool bNonDPassable=false;
813 
814  if("noop"!=VertexFunc){
815  if("out"==VertexFunc){
816  bOutputFound=true;
817  }else if("in"==VertexFunc) {
818  outSignal=EvaluateOutputs(v,g);
819  D.Dbg("1","outSignal==",outSignal.GetString());
820  processOutput(v,g,cv,outSignal,DLogic::X);
821  }else{
822  vector<DLogic> vSignals;
823  int totalDegree=degree(v,g); // Bug in in_degree
824  int numOutputs=out_degree(v,g);
825  int numInputs=totalDegree-numOutputs;
826 
827  InEdgeIteratorType startEI,endEI;
828  switch(numInputs){
829  case 0 :
830  break;
831  case 1 : // single input, see Note 1
832  tie(startEI,endEI)=in_edges(v,g);
833  outSignal=EvaluateSingleInput(VertexFunc,_e[*startEI]["label"]);
834  currentSignal=EvaluateOutputs(v,g);
835  bInconsistentOutput=processOutput(v,g,cv,outSignal,currentSignal);
836  break;
837  default : // multi-inputs
838  tie(startEI,endEI)=in_edges(v,g);
839  // see podem.oln Note1 : VC6 error if std:: left out of transform
840  // see podem.oln Note2 : VC6 error if std:: left out of back_inserter
841  std::transform(startEI,endEI,std::back_inserter(vSignals),GetSignalFunctor<G>(_g));
842  outSignal=EvaluateMultipleInputs(VertexFunc,vSignals);

844  if(false==DPassable(vSignals,outSignal)){
845  bNonDPassable=true;

847  }
848  bInconsistentOutput=processOutput(v,g,cv,outSignal,currentSignal);
849  break;
850  }//switch(numInputs)
851  }
852  }//if("noop"!=VertexFunc
853  return tripleBool(bOutputFound,bInconsistentOutput,bNonDPassable);
