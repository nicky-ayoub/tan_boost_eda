27 void CmdLine::process(int argc,char** argv){
28  if(debug==true) cout << "Processing input arguments\n";
29  for(int i=1;i<argc;i++){ // C++ skip command name itself, in argv[0]
30  if(debug==true){
31  cout << "\targv[" << i << "] " << argv[i] << "\n";
32  }
33  string flag=argv[i];
34  mapType::iterator pos=m.find(flag);
35 
36  if(pos==m.end()){
37  l.push_back(argv[i]);
38  }else if ("unset"==(*pos).second){
39  m.erase(pos);
40  m.insert(mapType::value_type(flag,"set"));
41  }else{
42  m.erase(pos);
43  m.insert(mapType::value_type(flag,argv[++i]));
44  }
45  }
46 }
